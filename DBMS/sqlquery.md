# creating a table named Menuitems with the following columns


CREATE TABLE MenuItems (
    item_id INT PRIMARY KEY,
    item_name VARCHAR(50) UNIQUE,
    Price INT NOT NULL,
    Category VARCHAR(20)
);

# Insert 3 new items into the menu.
INSERT INTO MenuItems
VALUES
    (1,'Pizza', 90, 'Main Course'),
    (2,'Burger', 50, 'Main Course'),
    (3,'Fries', 150, 'Side Dish');


# Update the price of an existing item in the menu.
UPDATE MenuItems SET Price = 100 WHERE item_id=1;

# Delete an item from the menu using item ID.
DELETE FROM MenuItems WHERE item_id=3;

# find the average price of all items in the menu.
SELECT AVG(price) AS AveragePrice FROM MenuItems;

# find the total revenue generated by all items in the menu.
SELECT SUM(price) AS TotalRevenue FROM MenuItems;
